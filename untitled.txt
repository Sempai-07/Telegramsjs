```js
public action(
    data: string,
    callback: (callbackQuery: CallbackQuery) => void,
    answer: boolean = false
    ): void {
    this.on('callback_query', (ctx) => {
      if (answer) {
        this.answerCallbackQuery({
          callbackQueryId: ctx.id
        })
      }
      if (ctx.data === data) {
        callback(ctx);
      }
    });
  }
```
